library(here)
library(tidyverse)

Import Data

ejscreen <- read_csv(here("data", "EJSCREEN_2024_BG_StatePct_with_AS_CNMI_GU_VI.csv"))

los_angeles <- ejscreen %>%
  filter(CNTY_NAME == "Los Angeles County") %>%
  select(ID:REGION, LIFEEXPPCT, PEOPCOLORPCT, LOWINCPCT, PM25, OZONE, PTSDF) %>%
  rename_all(tolower) %>%
  drop_na()
